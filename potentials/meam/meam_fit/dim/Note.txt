import math

# Define the values for F2, Cl2, Br2, I2
elements = {
    "F2": {"R0": 1.42, "E0": 1.60, "k": 890},
    "Cl2": {"R0": 1.99, "E0": 2.51, "k": 480},
    "Br2": {"R0": 2.28, "E0": 1.97, "k": 280},
    "I2": {"R0": 2.67, "E0": 1.54, "k": 230}
}

# Conversion factors
N_m_to_eV_A2 = 1 / (1.60218e-19 * 1e10)  # N/m to eV/Å²
GPa_to_eV_A3 = 1 / (1.60218e-19 * 1e30)  # GPa to eV/Å³

# Calculate alpha for each element
alpha_values = {}
for element, values in elements.items():
    R0 = values["R0"]
    E0 = values["E0"]
    k = values["k"] * N_m_to_eV_A2  # Convert force constant to eV/Å²
    Ki = 1  # Placeholder for bulk modulus in GPa, not provided in the data
    alpha = math.sqrt(Ki / E0) * R0
    alpha_values[element] = alpha

# Print the results
for element, alpha in alpha_values.items():
    print(f"{element}: alpha = {alpha:.15f} eV·Å")
